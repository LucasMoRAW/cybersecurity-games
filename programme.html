import hashlib

def hash_password(password):
    hash_object = hashlib.sha256(password.encode())
    hashed_password = hash_object.hexdigest()
    return hashed_password

user_password = "password123"
hashed = hash_password(user_password)
print("Mot de passe haché :", hashed)

import hashlib

def check_password(input_password, stored_hash):
    input_hash = hashlib.sha256(input_password.encode()).hexdigest()
    return input_hash == stored_hash

stored_password_hash = "6c8e8f2a4f68fd8b4013ba711db55e620c5c660d1565dbb1b06e8f1c6c8d61f3"
input_password = "password123"
if check_password(input_password, stored_password_hash):
    print("Mot de passe correct")
else:
    print("Mot de passe incorrect")

import ssl
import socket

def create_secure_connection(host, port):
    context = ssl.create_default_context()
    connection = context.wrap_socket(socket.socket(socket.AF_INET), server_hostname=host)
    connection.connect((host, port))
    return connection

secure_conn = create_secure_connection("www.example.com", 443)
print("Connexion sécurisée établie")
