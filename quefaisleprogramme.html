Voici un extrait de code Python. Quel est son rôle principal ?

import hashlib

def hash_password(password):
    # Génération d'un sel
    salt = "secure_salt"
    # Combinaison du mot de passe et du sel
    salted_password = password + salt
    # Calcul du hachage SHA-256
    hash_object = hashlib.sha256(salted_password.encode())
    hashed_password = hash_object.hexdigest()
    return hashed_password

# Exemple d'utilisation
user_password = "password123"
hashed = hash_password(user_password)
print("Mot de passe haché :", hashed)
