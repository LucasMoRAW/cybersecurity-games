Voici un extrait de code Python. Que fait ce programme ?

import base64

def encode_string(input_string):
    encoded = base64.b64encode(input_string.encode())
    return encoded.decode()

user_string = "hello_world"
encoded_string = encode_string(user_string)
print("Chaîne encodée en base64:", encoded_string)
